/* --- 追加：Markdown閲覧エンジン --- */ // 1. Marked.js を動的に読み込む const script = document.createElement('script'); script.src = "https://cdn.jsdelivr.net/npm/marked/marked.min.js"; document.head.appendChild(script); // 2. モーダル（閲覧窓）のHTMLを動的に生成して追加 const modalHTML = `
読み込み中...
`; document.body.insertAdjacentHTML('beforeend', modalHTML); // 3. mdファイルを取得して表示する関数 async function openMD(path) { const modal = document.getElementById('md-modal'); const content = document.getElementById('md-content'); modal.style.display = 'block'; gsap.from("#md-modal > div", { scale: 0.9, opacity: 0, duration: 0.4 }); try { const response = await fetch(path); const text = await response.text(); content.innerHTML = marked.parse(text); // ここでMarkdownをHTMLに変換 } catch (e) { content.innerHTML = "エラー: ファイルが見つかりません。"; } } function closeModal() { document.getElementById('md-modal').style.display = 'none'; } // 4. 各カードの onclick を書き換える（例：openMD('./principles/concept-of-quality.md')）
