<script>
//<![CDATA[  
(function() {
  document.addEventListener("DOMContentLoaded", function() {
    const tocContainer = document.getElementById("sticky-toc");
    // Materiapolloで一般的に使われる記事本文のセレクタ
    const postBody = document.querySelector(".post-body, .entry-content, #post-body");
    
    if (!tocContainer || !postBody) return;

    // 記事内のH2とH3を取得
    const headings = Array.from(postBody.querySelectorAll("h2, h3"));
    if (headings.length === 0) {
      document.getElementById("sticky-toc-container").style.display = "none";
      return;
    }

    const tocList = document.createElement("ul");

    // 目次項目の生成
    headings.forEach((heading, index) => {
      const id = "section-node-" + index;
      heading.id = id;
      
      const li = document.createElement("li");
      li.className = heading.tagName.toLowerCase() === "h3" ? "toc-h3" : "toc-h2";
      
      const a = document.createElement("a");
      a.href = "#" + id;
      a.className = "toc-link";
      a.textContent = heading.textContent;
      
      li.appendChild(a);
      tocList.appendChild(li);
    });
    tocContainer.appendChild(tocList);

    const tocLinks = document.querySelectorAll(".toc-link");

    // スクロール連動（ScrollSpy）の処理
    function updateActiveHeader() {
      const scrollPosition = window.scrollY + 120; // 判定位置をヘッダー分下げる
      let activeId = "";

      headings.forEach(heading => {
        if (heading.offsetTop <= scrollPosition) {
          activeId = heading.id;
        }
      });

      tocLinks.forEach(link => {
        link.classList.remove("active");
        if (link.getAttribute("href") === "#" + activeId) {
          link.classList.add("active");
        }
      });
    }

    // スクロールイベントを監視
    window.addEventListener("scroll", updateActiveHeader);
    // 初回実行
    updateActiveHeader();
  });
})();
//]]>
</script>  
  
  
<!-- 読了率 -->  
<script type='text/javascript'>
//<![CDATA[    
(function($) {
    $(window).on('scroll resize', function() {
        var $content = $('.post-body, .entry-content, .item-post-inner').first();
        if ($content.length === 0) return;

        var scrollTop = $(window).scrollTop();
        var contentTop = $content.offset().top;
        var contentHeight = $content.outerHeight();
        var windowHeight = $(window).height();

        // --- 透明度のコントロール ---
        if (scrollTop > 10) { // 10px以上スクロールしたら表示
            $('#reading-progress-container').addClass('is-scrolled');
        } else {
            $('#reading-progress-container').removeClass('is-scrolled');
        }

        // --- 読了率の計算 ---
        var progress = 0;
        // 記事のトップから計算を開始
        if (scrollTop > contentTop) {
            // 下記の200は、記事の終わり際で100%にするための調整値です
            progress = ((scrollTop - contentTop) / (contentHeight - windowHeight + 200)) * 100;
        }

        progress = Math.min(100, Math.max(0, progress));
        $('#reading-progress-bar').css('width', progress + '%');
    });
})(jQuery);
//]]>
</script>

(function() {
    const update = () => {
        // 1. HTMLを一切変更せず、既存の構造からターゲットを特定
        const container = document.querySelector('.radar-chart-2');
        if (!container) return;

        // 色（fill）を頼りに「塗りつぶしエリア」と「ドットのグループ」を見つける
        const poly = container.querySelector('path[fill="#d0582530"]');
        const dotsGroup = container.querySelector('g[fill="#d05825"]');
        const dds = container.querySelectorAll('dd');

        const centerX = 100, centerY = 100, radius = 100, count = dds.length;
        let points = [];

        // 2. ドットをクリアして再描画
        if (dotsGroup) dotsGroup.innerHTML = '';

        dds.forEach((dd, i) => {
            const val = parseFloat(dd.textContent) || 0;
            const angle = (Math.PI * 2 / count) * i - (Math.PI / 2);
            
            // 座標計算
            const x = centerX + (radius * (val / 10)) * Math.cos(angle);
            const y = centerY + (radius * (val / 10)) * Math.sin(angle);
            const px = x.toFixed(1), py = y.toFixed(1);
            
            points.push(px + " " + py);
            
            // ドット生成
            if (dotsGroup) {
                const circle = document.createElementNS("http://www.w3.org/2000/svg", "circle");
                circle.setAttribute("cx", px);
                circle.setAttribute("cy", py);
                circle.setAttribute("r", "3");
                dotsGroup.appendChild(circle);
            }
        });

        // 3. 塗りつぶしエリアの形状を更新
        if (poly) {
            poly.setAttribute("d", "M " + points.join(" L ") + " Z");
        }
    };

    // Bloggerの環境に合わせて確実に実行
    if (document.readyState === 'complete') {
        update();
    } else {
        window.addEventListener('load', update);
    }
})();
//]]>
</script>

<script type='text/javascript'>  
/*<![CDATA[*/
(function() {
  const path = window.location.pathname;
  const currentSearch = window.location.search;
  if (path.indexOf('.html') !== -1) return;

  const isArchive = /^\/\d{4}\/\d{2}\//.test(path);
  const maxResults = 7; 
  const dispPage = 2;

  function enforceLimit() {
    if (!isArchive) return;
    const allPosts = document.querySelectorAll('.post-outer');
    allPosts.forEach((post, index) => {
      if (index >= maxResults) {
        post.style.display = 'none';
        const dateOuter = post.closest('.date-outer');
        if (dateOuter) {
          const visible = Array.from(dateOuter.querySelectorAll('.post-outer')).filter(p => p.style.display !== 'none');
          if (visible.length === 0) dateOuter.style.display = 'none';
        }
      }
    });
  }

  if (isArchive && currentSearch.indexOf('max-results=') === -1) {
    let sep = (currentSearch.indexOf('?') === -1) ? '?' : '&';
    location.href = window.location.protocol + '//' + window.location.host + path + currentSearch + sep + 'max-results=' + maxResults + '&type=archive';
    return;
  }

  function getUrlParam(name) {
    const reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)");
    const r = currentSearch.substr(1).match(reg);
    return r != null ? decodeURIComponent(r[2]) : null;
  }

  let textLabel = path.includes('/search/label/') ? decodeURIComponent(path.split('/search/label/')[1].split(/[?#&]/)[0]) : "";
  let textQuery = getUrlParam('q');
  let archivePath = isArchive ? path.match(/^\/\d{4}\/\d{2}\//)[0] : "";
  let start = parseInt(getUrlParam('start')) || 0;
  let sumcnt = parseInt(getUrlParam('sumcnt')) || 0; 

  function getMonthBounds() {
    if (!isArchive) return null;
    const parts = archivePath.split('/');
    const startTime = parts[1] + "-" + parts[2] + "-01T00:00:00";
    let nextMonth = parseInt(parts[2]) + 1;
    let nextYear = parts[1];
    if (nextMonth > 12) { nextMonth = 1; nextYear = parseInt(parts[1]) + 1; }
    const endTime = nextYear + "-" + String(nextMonth).padStart(2, '0') + "-01T00:00:00";
    return { start: startTime, end: endTime };
  }

  function setHtmlPager(totalResults) {
    sumcnt = totalResults;
    let pagers = document.querySelectorAll('#blog-pager, .blog-pager, .paging-control');
    if (pagers.length === 0) {
      const main = document.querySelector('.blog-posts') || document.getElementById('Blog1');
      if (!main) return;
      const newPager = document.createElement('div');
      newPager.className = 'blog-pager custom-pager';
      newPager.id = 'blog-pager';
      main.parentNode.insertBefore(newPager, main.nextSibling);
      pagers = [newPager];
    }

    const totalPage = Math.ceil(sumcnt / maxResults);
    const currentPage = Math.floor(start / maxResults) + 1;

    let html = '';
    if (currentPage > 1) html += '<a href="#" class="pager-a" onclick="return pager(' + (currentPage - 1) + ')">＜</a>';
    for (let i = 1; i <= totalPage; i++) {
      if (i === currentPage) html += '<span id="pager-current">' + i + '</span>';
      else if (i === 1 || i === totalPage || Math.abs(i - currentPage) < dispPage)
        html += '<a href="#" class="pager-a" onclick="return pager(' + i + ')">' + i + '</a>';
      else if (Math.abs(i - currentPage) === dispPage) html += '<span class="pager-dots">...</span>';
    }
    if (currentPage < totalPage) html += '<a href="#" class="pager-a" onclick="return pager(' + (currentPage + 1) + ')">＞</a>';

    pagers.forEach(p => { 
      p.innerHTML = html; 
      p.style.display = (totalPage <= 1) ? 'none' : 'block';
    });
  }

  window.pager = function(page) {
    const skip = (page - 1) * maxResults;
    if (page === 1) {
      let dest = isArchive ? archivePath : (textLabel ? "/search/label/" + encodeURIComponent(textLabel) : "/");
      if (textQuery) dest = "/search?q=" + encodeURIComponent(textQuery);
      location.href = dest + (dest.indexOf('?') === -1 ? "?" : "&") + "max-results=" + maxResults + (isArchive ? "&type=archive" : "");
      return false;
    }
    
    const bounds = getMonthBounds();
    let feedUrl = (textLabel ? '/feeds/posts/summary/-/' + encodeURIComponent(textLabel) : '/feeds/posts/summary') +
                  '?alt=json&max-results=1&start-index=' + skip; 
    if (bounds) feedUrl += "&published-min=" + bounds.start + "&published-max=" + bounds.end;
    if (textQuery) feedUrl += "&q=" + encodeURIComponent(textQuery);

    fetch(feedUrl).then(r => r.json()).then(j => {
      if (!j.feed || !j.feed.entry) return;
      const rawDate = j.feed.entry[0].published.$t;
      const pubDate = encodeURIComponent(rawDate.substring(0,19) + rawDate.substring(23,29));
      let dest = textLabel ? "/search/label/" + encodeURIComponent(textLabel) : "/search";
      const queryPart = textQuery ? "&q=" + encodeURIComponent(textQuery) : "";
      
      location.href = dest + "?updated-max=" + pubDate + "&max-results=" + maxResults + "&start=" + skip + "&sumcnt=" + sumcnt + queryPart + (isArchive ? "&type=archive" : "");
    });
    return false;
  };

  function setIndexPager() {
    enforceLimit();
    if (sumcnt > 0) {
      setHtmlPager(sumcnt);
      return;
    }
    
    // 総件数取得のためのベースURL。qがある場合は検索フィードを使用
    let feedUrl = '/feeds/posts/summary?alt=json&max-results=0';
    const bounds = getMonthBounds();

    if (textLabel) {
      feedUrl = '/feeds/posts/summary/-/' + encodeURIComponent(textLabel) + '?alt=json&max-results=0';
    }
    
    if (isArchive && bounds) {
      feedUrl += "&published-min=" + bounds.start + "&published-max=" + bounds.end;
    }
    
    if (textQuery) {
      // 検索パラメータがある場合、max-results=0だと件数が正しく取れない場合があるため調整
      feedUrl = '/feeds/posts/summary?alt=json&max-results=50&q=' + encodeURIComponent(textQuery);
    }

    fetch(feedUrl).then(r => r.json()).then(json => {
      let totalResults = 0;
      if (json.feed && json.feed.openSearch$totalResults) {
        totalResults = parseInt(json.feed.openSearch$totalResults.$t);
      }
      
      // 特殊ケース：検索でmax-results以上のヒットがある場合、フィード内のentry数で補完
      if (textQuery && json.feed.entry && json.feed.entry.length > totalResults) {
        totalResults = json.feed.entry.length;
      }
      
      if (totalResults === 0) {
        const postsOnPage = document.querySelectorAll('.post-outer').length;
        if (postsOnPage >= maxResults) totalResults = postsOnPage + 1;
      }
      
      setHtmlPager(totalResults);
    }).catch(() => {
        const postsOnPage = document.querySelectorAll('.post-outer').length;
        if (postsOnPage > 0) setHtmlPager(postsOnPage >= maxResults ? maxResults + 1 : postsOnPage);
    });

    if (currentPath.indexOf(EXCLUDE_PATH) !== -1 ||

        currentUrl.indexOf("draft.blogger.com") !== -1) {
      return;
    }

    let maxScrollRate = 0;

    window.addEventListener("scroll", function() {
      const docHeight = document.documentElement.scrollHeight - window.innerHeight;
      const currentRate = docHeight > 0 ? window.scrollY / docHeight : 0;
      if (currentRate > maxScrollRate) maxScrollRate = currentRate;
    }, { passive: true });

    window.addEventListener("visibilitychange", function() {

        const stayTimeSec = (Date.now() - startTime) / 1000;

          path: currentPath,
          title: fullTitle,
          score: stayTimeSec * Math.max(maxScrollRate, 0.1),
          count: 1,
          stayTime: stayTimeSec
        };

        const blob = new Blob([JSON.stringify(payload)], {type: 'text/plain'});
        navigator.sendBeacon(GAS_URL, blob);
      }
    });
  })();

    fetch(GAS_URL)
      .then(response => response.json())
      .then(data => {
